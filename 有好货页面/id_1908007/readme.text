// jsx的核心代码
import {Carousel} from "./carousel.js";

function myCreate(Class, attributes){
    var object = new Class();
    for(let name in attributes)
        object.setAttribute(name, attributes[name]);
    return object; 
}

</Carousel>
c.appendTo(document.body);
